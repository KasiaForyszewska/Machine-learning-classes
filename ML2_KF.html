#Klasa dedykowana regresji liniowej 2 zmiennych

class LinearModel2v:
    def __init__(self, eta = 0.001, diff = 0.001, w1 = 1, w2 = 1, w0 = 1, maxiter = 10000):
        self.eta = eta
        self.diff = diff
        self.w1 = w1
        self.w2 = w2
        self.w0 = w0
        self.maxiter = maxiter
        
    def loss_function(self, X, t):
        N = len(X)
        C = 0
        for i in range(N):
            C += (X[i][0]*self.w1+X[i][1]*self.w2+self.w0-t[i])**2
        return C/(2*N)
    
    def update_weights(self, X, t):
        N = len(X)
        dC1 = 0
        dC2 = 0
        dC0 = 0
        for i in range(N):
            y_pred = X[i][0]*self.w1+X[i][1]*self.w2+self.w0
            dC1 += 2*X[i][0]*(y_pred-t[i])
            dC2 += 2*X[i][1]*(y_pred-t[i])
            dC0 += 2*(y_pred-t[i])

        self.w1 = self.w1 - self.eta*dC1/(2*N)
        self.w2 = self.w2 - self.eta*dC2/(2*N)
        self.w0 = self.w0 - self.eta*dC0/(2*N)

    def train(self, X, t):
        l = []
        ile = 0
        while True:
            l.append(self.loss_function(X, t))
            self.update_weights(X, t)
            ile += 1
            if len(l)>2:
                if abs(l[-1]-l[-2])/l[-1] <  self.diff or ile > self.maxiter:
                    break

Regularyzacja

Wprowadzenie szerszej klasy funkcji bazowych jak i iloci parametrw moe bardzo szybko prowadzi do zjawiska overfittingu (przetrenowania). Jak sobie z tym poradzi? Na przykad, wprowadzajc pewne dodatkowe obcienie do funkcji kosztu zalene od wspczynnikw:

Regularyzacja L2 (Ridge Regression): $LF^+ = \alpha \sum_{i} w_{i}^2$

Regularyzacja L1 (Lasso Regression): $LF^+ = \alpha \sum_{i} |w_{i}|^2$

tzn. $LF_{new} = LF + LF^+$

class ModelRidge:
    def __init__(self, eta = 0.001, diff = 0.001, w1 = 1, w2 = 1, w0 = 1, maxiter = 10000):
        self.eta = eta
        self.diff = diff
        self.w1 = w1
        self.w2 = w2
        self.w0 = w0
        self.maxiter = maxiter
        
    def loss_function(self, X, t):
        N = len(X)
        C = 0
        alpha = 0.5
        for i in range(N):
            lr = (X[i][0]*self.w1+X[i][1]*self.w2+self.w0-t[i])**2
            rr = alpha*(self.w1**2+self.w2**2+self.w0**2)
            C += lr
            LR = lr/(2*N)

,,model.coef" i ,,model.intercept" su do wywoania wartoci wag i bias. Wagi kontroluj sygna (lub si poczenia) pomidzy neuronami. Decyduj one o istotnoci danych wejciowych. Drugi z parametrw jest stay, stanowi dodatkowy parametr wejciowy do kolejnej warstwy i gwarantuje przepyw sygnau na neuronie, nawet gdy dane wejciowe s zerami.

Metoda ,,predict" zwraca przewidywane wartoci etykiet danych przekazanych jako argument na podstawie wyuczonych lub wytrenowanych danych uzyskanych z modelu. Metoda ,,predict_proba" zwraca prawdopodobiestwa klas dla kadego punktu danych. Metoda przyjmuje pojedynczy argument odpowiadajcy danym, dla ktrych zostan obliczone prawdopodobiestwa, i zwraca tablic list zawierajcych prawdopodobiestwa klas dla wejciowych punktw danych. Reasumujc, pierwsza z metod suy do przewidywania rzeczywistej klasy, podczas gdy druga moe by uywana do wywnioskowania prawdopodobiestw klas (tj. prawdopodobiestwa, e dany punkt danych naley do klas bazowych).

Parametr C to odwrotno siy regularyzacji. Mniejsze jego wartoci okrelaj silniejsz regularyzacj. Tak wic jest to wykres obrazujcy zaleno midzy wartoci wag a si regularyzacji. Spadek wagi nie ma wikszego wpywu na regularyzacj w przypadku proantocyjanw oraz magnezu. Wraz ze spadkiem wagi, sia regularyzacji ronie najszybciej dla intensywnoci koloru. Na kolejnych miejsach uplasoway si prolina i ash. Natomiast, wraz ze wzrostem wagi sia regularyzacji najszybciej maleje w przypadku flawonoidw, nastpnie zasadowo popiou oraz odcie.
